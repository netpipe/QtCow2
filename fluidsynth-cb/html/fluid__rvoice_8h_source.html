202 /*
203  * Combines the most significant 16 bit part of a sample with a potentially present
204  * least sig. 8 bit part in order to create a 24 bit sample.
205  */
206 static FLUID_INLINE int32_t
207 fluid_rvoice_get_sample(const short int* dsp_msb, const char* dsp_lsb, unsigned int idx)
208 {
209  /* cast sample to unsigned type, so we can safely shift and bitwise or
210  * without relying on undefined behaviour (should never happen anyway ofc...) */
211  uint32_t msb = (uint32_t)dsp_msb[idx];
212  uint8_t lsb = 0U;
213 
214  /* most soundfonts have 16 bit samples, assume that it's unlikely we
215  * experience 24 bit samples here */
216  if(FLUID_UNLIKELY(dsp_lsb != NULL))
217  {
218  lsb = (uint8_t)dsp_lsb[idx];
219  }
220 
221  return (int32_t)((msb << 8) | lsb);
222 }
