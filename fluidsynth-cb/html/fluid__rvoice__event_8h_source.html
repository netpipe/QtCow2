60 static FLUID_INLINE void

62 {
63  int queue_stored = fluid_atomic_int_get(&handler->queue_stored);
64 
65  if (queue_stored > 0) {
66  fluid_atomic_int_set(&handler->queue_stored, 0);
67  fluid_ringbuffer_next_inptr(handler->queue, queue_stored);
